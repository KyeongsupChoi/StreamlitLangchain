# Multi-Agent: Router Pattern

The router pattern classifies an incoming query, dispatches it to one or more specialized agents in parallel, and synthesizes their results. This is ideal for multi-source knowledge bases where different sources require different retrieval strategies.

---

## Architecture

```
User Query -> Classification -> [Send to matching agents in parallel]
                                      |          |          |
                                   Agent A    Agent B    Agent C
                                      |          |          |
                                      +-----+----+----------+
                                            |
                                      Synthesize Results
                                            |
                                      Final Answer
```

---

## State Definitions

```python
from typing import TypedDict, Annotated, Literal
from langchain_core.messages import AnyMessage
from langgraph.graph import add_messages
import operator

class AgentInput(TypedDict):
    """Input to a specialized agent."""
    query: str
    source: str

class AgentOutput(TypedDict):
    """Output from a specialized agent."""
    source: str
    answer: str
    confidence: float

class Classification(TypedDict):
    """Result of query classification."""
    sources: list[str]
    reasoning: str

class RouterState(TypedDict):
    """Overall state for the router workflow."""
    messages: Annotated[list[AnyMessage], add_messages]
    query: str
    classification: Classification
    agent_outputs: Annotated[list[AgentOutput], operator.add]  # Reducer: concatenate lists
    final_answer: str
```

The `agent_outputs` field uses `operator.add` as a reducer so that outputs from parallel agents are merged into a single list.

## Structured Output for Classification

Use Pydantic to get structured classification from the LLM:

```python
from pydantic import BaseModel, Field
from langchain_openai import ChatOpenAI

class ClassificationResult(BaseModel):
    """Classification of which knowledge sources are relevant."""
    sources: list[Literal["product_docs", "api_reference", "community_forum", "internal_wiki"]] = Field(
        description="List of relevant knowledge sources for this query"
    )
    reasoning: str = Field(
        description="Brief explanation of why these sources were selected"
    )

llm = ChatOpenAI(model="gpt-4o")
structured_llm = llm.with_structured_output(ClassificationResult)

def classify_query(state: RouterState) -> dict:
    """Classify the query to determine which sources to search."""
    query = state["query"]

    result = structured_llm.invoke([
        {
            "role": "system",
            "content": (
                "Classify which knowledge sources are most relevant for answering this query. "
                "Choose one or more sources:\n"
                "- product_docs: Official product documentation and guides\n"
                "- api_reference: API endpoints, parameters, and code examples\n"
                "- community_forum: User discussions, workarounds, and tips\n"
                "- internal_wiki: Internal team knowledge and architecture docs"
            )
        },
        {"role": "user", "content": query}
    ])

    return {
        "classification": {
            "sources": result.sources,
            "reasoning": result.reasoning,
        }
    }
```

## Specialized Agents

Each agent handles one knowledge source:

```python
def search_product_docs(state: RouterState) -> dict:
    """Search official product documentation."""
    query = state["query"]
    # In production, this would search a real vector store
    answer = llm.invoke([
        {"role": "system", "content": "You are a product documentation expert. Answer based on official docs."},
        {"role": "user", "content": query}
    ]).content

    return {
        "agent_outputs": [{
            "source": "product_docs",
            "answer": answer,
            "confidence": 0.9,
        }]
    }
```

## Send API for Parallel Fan-Out

The `Send` API dispatches work to multiple nodes in parallel:

```python
from langgraph.constants import Send

def route_to_agents(state: RouterState) -> list[Send]:
    """Route to the appropriate agents based on classification."""
    sources = state["classification"]["sources"]
    sends = []
    for source in sources:
        if source in SOURCE_AGENTS:
            sends.append(Send(source, state))
    return sends
```

## Synthesize Results

Combine all agent outputs into a single coherent answer:

```python
def synthesize_results(state: RouterState) -> dict:
    """Synthesize results from all agents into a final answer."""
    query = state["query"]
    agent_outputs = state["agent_outputs"]

    if not agent_outputs:
        return {"final_answer": "I couldn't find relevant information for your query."}

    # Sort by confidence
    sorted_outputs = sorted(agent_outputs, key=lambda x: x["confidence"], reverse=True)

    # Format sources for the synthesis prompt
    sources_text = "\n\n".join(
        f"[{out['source']}] (confidence: {out['confidence']}):\n{out['answer']}"
        for out in sorted_outputs
    )

    synthesis = llm.invoke([
        {
            "role": "system",
            "content": (
                "Synthesize the following information from multiple sources into a single, "
                "coherent answer. Cite which source each piece of information comes from. "
                "Prioritize higher-confidence sources."
            )
        },
        {
            "role": "user",
            "content": f"Question: {query}\n\nSources:\n{sources_text}"
        }
    ]).content

    return {
        "final_answer": synthesis,
        "messages": [{"role": "assistant", "content": synthesis}]
    }
```

## Building the Graph

```python
from langgraph.graph import StateGraph, START, END

builder = StateGraph(RouterState)

# Add nodes
builder.add_node("classify", classify_query)
builder.add_node("product_docs", search_product_docs)
builder.add_node("api_reference", search_api_reference)
builder.add_node("community_forum", search_community_forum)
builder.add_node("internal_wiki", search_internal_wiki)
builder.add_node("synthesize", synthesize_results)

# Edges
builder.add_edge(START, "classify")
builder.add_conditional_edges("classify", route_to_agents)

# All agent nodes feed into synthesize
builder.add_edge("product_docs", "synthesize")
builder.add_edge("api_reference", "synthesize")
builder.add_edge("community_forum", "synthesize")
builder.add_edge("internal_wiki", "synthesize")

builder.add_edge("synthesize", END)

graph = builder.compile()
```

### Running the Router

```python
result = graph.invoke({
    "query": "How do I authenticate API requests?",
    "messages": [{"role": "user", "content": "How do I authenticate API requests?"}],
    "agent_outputs": [],
})

print("Classification:", result["classification"])
print("\nFinal Answer:", result["final_answer"])
print(f"\nSearched {len(result['agent_outputs'])} sources")
```

## Advanced: Stateful Routers

### Tool Wrapper Approach

Wrap the entire router as a tool for use inside a conversational agent:

```python
from langchain.tools import tool

@tool
def knowledge_search(query: str) -> str:
    """Search across all knowledge sources for information."""
    result = graph.invoke({
        "query": query,
        "messages": [],
        "agent_outputs": [],
    })
    return result["final_answer"]

# Use in a conversational agent
conversational_agent = create_react_agent(
    model=llm,
    tools=[knowledge_search],
    prompt="You are a helpful assistant. Use knowledge_search for any factual questions.",
)
```

### Full Persistence

Add checkpointing for multi-turn conversations with the router:

```python
from langgraph.checkpoint.memory import MemorySaver

checkpointer = MemorySaver()
stateful_graph = builder.compile(checkpointer=checkpointer)

config = {"configurable": {"thread_id": "session-1"}}

# First query
result = stateful_graph.invoke(
    {
        "query": "How do I authenticate?",
        "messages": [{"role": "user", "content": "How do I authenticate?"}],
        "agent_outputs": [],
    },
    config=config,
)

# Follow-up query in same session
result = stateful_graph.invoke(
    {
        "query": "Show me a code example for that",
        "messages": [{"role": "user", "content": "Show me a code example for that"}],
        "agent_outputs": [],
    },
    config=config,
)
```
